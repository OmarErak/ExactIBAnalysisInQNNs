\documentclass[crop,tikz]{standalone}% 'crop' is the default for v1.0, before it was 'preview'
\usepackage{pgfplots}
\usepackage{pgfplotstable}
\usepgfplotslibrary{groupplots}

\input{setup.tex}

\begin{document}
\pgfplotstableread[col sep=comma]{quantize-extra/SYN-Tanh/4/accuracy.csv}\tabTQ
\pgfplotstableread[col sep=comma]{quantize-extra/SYN-ReLU/4/accuracy.csv}\tabRQ

\usepgfplotslibrary{fillbetween}
\pgfplotsset{
compat=1.16,
acc axis/.style={
	xmin=0,xmax=8000,
	ymin=0.8,ymax=1,
	legend style={ at={(1.5,0.33)}, anchor=south east},
	xlabel={Epoch},
	ylabel={Accuracy},
},
}
\begin{tikzpicture}
\begin{groupplot}[group style={group size= 1 by 1}]
	\nextgroupplot[acc axis,title=4-bit Quantized Networks]
		% Legend
		\addlegendimage{no markers,colTanh}
		\addlegendentry{\tanh\ (train)}
		\addlegendimage{no markers,dashed,colTanh}
		\addlegendentry{\tanh\ (test)}
		\addlegendimage{no markers,colRelu}
		\addlegendentry{\relu\ (train)}
		\addlegendimage{no markers,dashed,colRelu}
		\addlegendentry{\relu\ (test)}

		% Tanh train
		\addplot[color=colTanh] table[x=epoch,y=train_mean]{\tabTQ};
		\addplot[name path=tanh_up, color=colTanh!40] table[x=epoch,y=train_up]{\tabTQ};
		\addplot[name path=tanh_lw, color=colTanh!40] table[x=epoch,y=train_lw]{\tabTQ};
		\addplot[colTanh!40,fill opacity=0.4] fill between[of=tanh_up and tanh_lw];
		
		% Tanh test
		\addplot[color=colTanh,dashed] table[x=epoch,y=test_mean]{\tabTQ};
		\addplot[name path=tanh_up2, color=colTanh!40] table[x=epoch,y=test_up]{\tabTQ};
		\addplot[name path=tanh_lw2, color=colTanh!40] table[x=epoch,y=test_lw]{\tabTQ};
		\addplot[colTanh!40,fill opacity=0.4] fill between[of=tanh_up2 and tanh_lw2];
		
		% Relu train
		\addplot[color=colRelu] table[x=epoch,y=train_mean]{\tabRQ};
		\addplot[name path=relu_up, color=colRelu!40] table[x=epoch,y=train_up]{\tabRQ};
		\addplot[name path=relu_lw, color=colRelu!40] table[x=epoch,y=train_lw]{\tabRQ};
		\addplot[colRelu!40,fill opacity=0.4] fill between[of=relu_up and relu_lw];
		
		% Relu test
		\addplot[color=colRelu,dashed] table[x=epoch,y=test_mean]{\tabRQ};
		\addplot[name path=relu_up2, color=colRelu!40] table[x=epoch,y=test_up]{\tabRQ};
		\addplot[name path=relu_lw2, color=colRelu!40] table[x=epoch,y=test_lw]{\tabRQ};
		\addplot[colRelu!40,fill opacity=0.4] fill between[of=relu_up2 and relu_lw2];
\end{groupplot}
\end{tikzpicture}
\end{document}
